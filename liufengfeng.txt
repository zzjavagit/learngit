#对deposite表进行统计，按银行统计存款总数，显示为b_id,total
SELECT b.b_id,SUM(d.amount) FROM deposite d,bank b WHERE 
b.b_id=d.b_id GROUP BY d.b_id

#对deposite、customer、bank进行查询，查询条件为location在广州、苏州、济南的客户，
#存款在300000至500000之间的存款记录，显示客户姓名name、银行名称bank_name、存款金额amount.
SELECT c.name,b.bank_name,d.amount FROM deposite d,customer c,bank b WHERE 
d.c_id=c.c_id AND d.b_id=b.b_id AND c.location IN('广州','苏州','济南') and d.amount>300000 and d.amount<500000
